```typescript
import { create } from 'zustand';
import * as environmentsApi from '../api/environments';
import type { EnvironmentSummary } from '../types/environments';
import { load } from '@tauri-apps/plugin-store';

interface EnvironmentStore {
  environments: EnvironmentSummary[];
  activeEnvSlug: string | null;
  activeEnvVariables: Record<string, string>; // merged public + secret values

  loadEnvironments: (workspaceId: string) => Promise<void>;
  setActiveEnvironment: (workspaceId: string, slug: string | null) => Promise<void>;
  createEnvironment: (workspaceId: string, name: string) => Promise<void>;
  deleteEnvironment: (workspaceId: string, slug: string) => Promise<void>;
  renameEnvironment: (workspaceId: string, oldSlug: string, newName: string) => Promise<void>;
  refreshActiveVariables: (workspaceId: string) => Promise<void>;
}
```

Implementation details:
- `loadEnvironments`: calls `environmentsApi.listEnvironments(workspaceId)`, sets `environments`. Then loads persisted `activeEnvSlug` from tauri-plugin-store using `const store = await load('dispatch-prefs.json', { autoSave: true }); const saved = await store.get<string | null>(\`activeEnv:${workspaceId}\`);`. If saved slug exists in the loaded environments list, calls `refreshActiveVariables`. Otherwise clears active env.
- `setActiveEnvironment`: sets `activeEnvSlug`, persists to store via `store.set(\`activeEnv:${workspaceId}\`, slug)`, if slug is not null calls `refreshActiveVariables`, otherwise sets `activeEnvVariables: {}`.
- `refreshActiveVariables`: loads the full environment file via `environmentsApi.loadEnvironment(workspaceId, activeEnvSlug)`, loads secrets via `environmentsApi.loadSecretValues(workspaceId, activeEnvSlug)`, merges: for each variable, if `variable.secret` use the secret value from the secrets map, else use `variable.value`. Sets `activeEnvVariables` to the merged `Record<string, string>`.
- `createEnvironment`: calls API, then calls `loadEnvironments` to refresh list.
- `deleteEnvironment`: calls API. If deleted slug === activeEnvSlug, clears active env. Then calls `loadEnvironments`.
- `renameEnvironment`: calls API, then calls `loadEnvironments`.

**4. Create `src/utils/variables.ts`:**
```typescript
export type VariableToken =
  | { type: 'literal'; text: string }
  | { type: 'variable'; text: string; key: string; resolved: boolean };

const VAR_REGEX = /\{\{([a-zA-Z_][a-zA-Z0-9_]*)\}\}/g;

export function tokenize(input: string, vars: Record<string, string>): VariableToken[] {
  const tokens: VariableToken[] = [];
  let last = 0;
  for (const m of input.matchAll(VAR_REGEX)) {
    if (m.index! > last) {
      tokens.push({ type: 'literal', text: input.slice(last, m.index!) });
    }
    tokens.push({
      type: 'variable',
      text: m[0],
      key: m[1],
      resolved: m[1] in vars,
    });
    last = m.index! + m[0].length;
  }
  if (last < input.length) {
    tokens.push({ type: 'literal', text: input.slice(last) });
  }
  return tokens;
}

export function substitute(s: string, vars: Record<string, string>): string {
  return s.replace(VAR_REGEX, (_, key) => vars[key] ?? `{{${key}}}`);
}

export function extractVariables(s: string): string[] {
  return [...s.matchAll(VAR_REGEX)].map(m => m[1]);
}

export function countUnresolved(fields: string[], vars: Record<string, string>): number {
  const all = fields.flatMap(extractVariables);
  return new Set(all.filter(k => !(k in vars))).size;
}
```
