  it('returns null for non-curl input', () => {

    const result = buildCurlString('GET', 'https://example.com/api', [], null, null, {});

    expect(result).toContain('"https://example.com/api"');

**Step 3: Create `src/utils/curl.ts` (GREEN phase).**

```typescript
import { toJsonString } from 'curlconverter';
import { substitute } from './variables';
import type { KeyValueEntry, RequestBody, RequestAuth } from '../types/collections';

export interface ParsedCurl {
  method: string;
  url: string;
  headers: Record<string, string>;
  body: string | null;
}

export function parseCurl(curlCommand: string): ParsedCurl | null {
  if (!curlCommand || !curlCommand.trimStart().startsWith('curl ')) {
    return null;
  }
  try {
    const json = JSON.parse(toJsonString(curlCommand));
    return {
      method: (json.method ?? 'GET').toUpperCase(),
      url: json.url ?? '',
      headers: json.headers ?? {},
      body: typeof json.data === 'string' ? json.data
        : json.data ? JSON.stringify(json.data) : null,
    };
  } catch {
    return null;
  }
}

export function buildCurlString(
  method: string,
  url: string,
  headers: KeyValueEntry[],
  body: RequestBody | null,
  auth: RequestAuth | null,
  vars: Record<string, string>,
): string {
  const resolvedUrl = substitute(url, vars);
  const parts: string[] = [`curl -X ${method} "${resolvedUrl}"`];

  if (auth?.token) {
    parts.push(`  -H "Authorization: Bearer ${substitute(auth.token, vars)}"`);
  }

  if (body?.type === 'json' && body.content) {
    parts.push('  -H "Content-Type: application/json"');
  }

  for (const h of headers.filter((h) => h.enabled)) {
    parts.push(`  -H "${substitute(h.key, vars)}: ${substitute(h.value, vars)}"`);
  }

  if (body?.content) {
    const data = substitute(body.content, vars);
    parts.push(`  -d '${data.replace(/'/g, "'\\''")}'`);
  }

  return parts.join(' \\\n');
}
```

**Step 2: Create an `applyParsedCurl` helper** (can be inline in UrlBar or a shared function).

```typescript
function applyParsedCurl(parsed: ParsedCurl) {
  const store = useRequestStore.getState();
  store.setMethod(parsed.method);
  store.setUrl(parsed.url);
  store.setHeaders(
    Object.entries(parsed.headers).map(([key, value]) => ({ key, value, enabled: true }))
  );
  if (parsed.body) {
    store.setBody({ type: 'json', content: parsed.body });
  } else {
    store.setBody(null);
  }
  // Check for Authorization Bearer header and extract to auth
  const authHeader = Object.entries(parsed.headers).find(
    ([k]) => k.toLowerCase() === 'authorization'
  );
  if (authHeader && authHeader[1].startsWith('Bearer ')) {
    store.setAuth({ type: 'bearer', token: authHeader[1].slice(7) });
    // Remove Authorization from headers since it's now in auth
    store.setHeaders(
      Object.entries(parsed.headers)
        .filter(([k]) => k.toLowerCase() !== 'authorization')
        .map(([key, value]) => ({ key, value, enabled: true }))
    );
  }
}
```

const handlePaste = (e: React.ClipboardEvent<HTMLInputElement>) => {
  const text = e.clipboardData.getData('text');
  if (text.trimStart().startsWith('curl ')) {
