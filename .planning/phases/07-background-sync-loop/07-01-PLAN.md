```rust
#[tauri::command]
#[specta::specta]
pub async fn notify_change(
    app: tauri::AppHandle,
    workspace_id: String,
) -> Result<(), String> {
    use tauri::Manager;
    let entries = crate::workspace::registry::load_registry(&app)?;
    let entry = entries.iter().find(|e| e.id == workspace_id)
        .ok_or_else(|| "workspace not found".to_string())?.clone();
    if entry.is_local { return Ok(()); } // Skip local workspaces
    let clone_url = entry.clone_url.ok_or("no clone url")?;
    let token = crate::auth::token::load_token(&app)?
        .ok_or_else(|| "not_authenticated".to_string())?;
    let actor = app.state::<crate::sync::ActorHandle>();
    actor.notify_change(workspace_id, entry.local_path, clone_url, token).await
}
```

**Step 1: Create a shared `notify_change_inner` helper** in `src-tauri/src/commands/sync.rs` (or a utility function). This avoids code duplication across command files. The helper:
```rust
pub(crate) async fn notify_change_inner(app: &tauri::AppHandle, workspace_id: String) {
    use tauri::Manager;
    // Best-effort: any failure is silently ignored (sync failure should not fail the save)
    let Ok(entries) = crate::workspace::registry::load_registry(app) else { return; };
    let Some(entry) = entries.iter().find(|e| e.id == workspace_id).cloned() else { return; };
    if entry.is_local { return; }
    let Some(clone_url) = entry.clone_url else { return; };
    let Ok(Some(token)) = crate::auth::token::load_token(app) else { return; };
    let actor = app.state::<crate::sync::ActorHandle>();
    let _ = actor.notify_change(workspace_id, entry.local_path, clone_url, token).await;
}
```

- After `app.manage(sync::ActorHandle::new(app_handle_clone));` (note: ActorHandle::new now takes AppHandle from Task 1).

- Spawn the periodic pull task:
```rust
{
    let periodic_actor = app.state::<sync::ActorHandle>().inner().clone();
    let periodic_app = app.handle().clone();
    tauri::async_runtime::spawn(async move {
        use tokio::time::{interval, Duration};
        let mut ticker = interval(Duration::from_secs(30));
        ticker.tick().await; // consume immediate first tick
        loop {
            ticker.tick().await;
            let Ok(entries) = workspace::registry::load_registry(&periodic_app) else { continue; };
            for entry in entries.iter().filter(|e| !e.is_local && e.clone_url.is_some()) {
                let Ok(Some(token)) = auth::token::load_token(&periodic_app) else { continue; };
                let _ = periodic_actor.pull(
                    entry.id.clone(),
                    entry.local_path.clone(),
                    entry.clone_url.clone().unwrap(),
                    token,
                ).await;
            }
        }
    });
}
```
